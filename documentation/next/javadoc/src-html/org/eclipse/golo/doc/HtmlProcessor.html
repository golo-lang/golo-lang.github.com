011package org.eclipse.golo.doc;
012
013import org.eclipse.golo.compiler.parser.ASTCompilationUnit;
014import gololang.FunctionReference;
015import gololang.IO;
016
017import java.nio.file.Path;
018import java.util.Map;
019import java.util.List;
020
021
022import com.github.rjeschke.txtmark.BlockEmitter;
023import com.github.rjeschke.txtmark.Configuration;
024import com.github.rjeschke.txtmark.Processor;
025
026
027public class HtmlProcessor extends AbstractProcessor {
028
029  private Path srcFile;
030  private final DocIndex globalIndex = new DocIndex();
031
032  @Override
033  protected String fileExtension() {
034    return "html";
035  }
036
037  public DocIndex globalIndex() {
038    return globalIndex;
039  }
040
041  /**
042   * Returns the direct link to the given documentation element from a given filename.
043   */
044  public String linkToDoc(String src, DocumentationElement dst) {
045    Path out = outputFile(src);
046    if (out.getParent() != null) {
047      out = out.getParent();
048    }
049    return out.relativize(docFile(dst)).toString()
050      + (dst.id().isEmpty() ? "" : ("#" + dst.id()));
051  }
052
053  @Override
054  public String render(ModuleDocumentation documentation) throws Throwable {
055    FunctionReference template = template("template", fileExtension());
056    globalIndex.update(documentation);
057    addModule(documentation);
058    Path doc = docFile(documentation);
059    if (doc.getParent() != null) {
060      doc = doc.getParent();
061    }
062    return (String) template.invoke(this, documentation, doc.relativize(srcFile));
063  }
064
065  @Override
066  public void process(Map<String, ModuleDocumentation> docs, Path targetFolder) throws Throwable {
067    setTargetFolder(targetFolder);
068    for (Map.Entry<String, ModuleDocumentation> doc : docs.entrySet()) {
069      renderModule(doc.getKey(), doc.getValue());
070    }
071    renderIndex("index");
072    renderIndex("index-all");
073  }
074
075  private void renderModule(String sourceFile, ModuleDocumentation doc) throws Throwable {
076    String moduleName = doc.moduleName();
077    srcFile = outputFile(moduleName + "-src");
078    IO.textToFile(renderSource(moduleName, sourceFile), srcFile);
079    IO.textToFile(render(doc), outputFile(moduleName));
080  }
081
082  private String renderSource(String moduleName, String filename) throws Throwable {
083    FunctionReference template = template("src", fileExtension());
084    String content = (String) IO.fileToText(filename, "UTF-8");
085    int nbLines = 0;
086    for (int i = 0; i < content.length(); i++) {
087      if (content.charAt(i) == '\n') {
088        nbLines++;
089      }
090    }
091    return (String) template.invoke(moduleName, content, nbLines);
092  }
093
094  public static BlockEmitter blockHighlighter() {
095    return new BlockEmitter() {
096      @Override
097      public void emitBlock(StringBuilder out, List<String> lines, String meta) {
098        String language;
099        if ("".equals(meta)) {
100          language = "golo";
101        } else {
102          language = meta;
103        }
104        out.append("<pre class=\"listing\">");
105        out.append(String.format("<code class=\"lang-%s\" data-lang=\"%s\">", language, language));
106        for (String rawLine : lines) {
107          String line = rawLine
108            .replace("&", "&amp;")
109            .replace(">", "&gt;")
110            .replace("<", "&lt;");
111          out.append(line);
112          out.append('\n');
113        }
114        out.append("</code></pre>");
115        out.append('\n');
116      }
117    };
118  }
119
120  public static String sectionTitle(int level, DocumentationElement doc, Path src) {
121    String permalink = String.format("<a class=\"permalink\" href=\"#%s\" title=\"link to this section\">&#182;</a>",
122        doc.id());
123    String srclink = src == null ? ""
124      : String.format("<nav class=\"srclink\"><a href=\"%s#l-%s\" rel=\"source\" title=\"Link to the corresponding source\">Source</a></nav>",
125          src, doc.line());
126    return String.format("<h%s id=\"%s\">%s%s</h%s>%s", level, doc.id(), doc.label(), permalink, level, srclink);
127  }
128
129  public static String tocItem(DocumentationElement doc) {
130    return String.format("<a href=\"#%s\">%s</a>", doc.id(), doc.label());
131  }
132
133  public static String process(String documentation, int rootLevel, Configuration configuration) {
134    return Processor.process(AbstractProcessor.adaptSections(documentation, rootLevel), configuration);
135  }
136}
