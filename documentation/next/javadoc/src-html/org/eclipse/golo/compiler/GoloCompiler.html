011package org.eclipse.golo.compiler;
012
013import org.eclipse.golo.compiler.ir.GoloModule;
014import org.eclipse.golo.compiler.parser.ASTCompilationUnit;
015import org.eclipse.golo.compiler.parser.GoloOffsetParser;
016import org.eclipse.golo.compiler.parser.GoloParser;
017import org.eclipse.golo.compiler.parser.ParseException;
018
019import java.io.*;
020import java.nio.charset.Charset;
021import java.nio.charset.UnsupportedCharsetException;
022import java.util.Collections;
023import java.util.List;
024import java.util.jar.JarOutputStream;
025import java.util.zip.ZipEntry;
026
027import static gololang.Messages.message;
028
029/**
030 * The Golo compiler.
031 * <p>
032 * Instances of this class may be reused to compile several sources.
033 * <p>
034 * Several methods are made public while they do not necessarily need so for the needs of the Golo compiler.
035 * Such deviations from a "good and clean" design are on-purpose, as this facilitates the implementation of
036 * Golo support in IDEs.
037 */
038public class GoloCompiler {
039
040  private GoloParser parser;
041  private GoloCompilationException.Builder exceptionBuilder = null;
042
043
044  /**
045   * Initializes an ExceptionBuilder to collect errors instead of throwing immediately.
046   * This method is made public for the requirements of IDEs support.
047   *
048   * @param builder the exception builder to add problems into.
049   */
050  public final void setExceptionBuilder(GoloCompilationException.Builder builder) {
051    exceptionBuilder = builder;
052  }
053
054  private GoloCompilationException.Builder getOrCreateExceptionBuilder(String goloSourceFile) {
055    if (exceptionBuilder == null) {
056      exceptionBuilder = new GoloCompilationException.Builder(goloSourceFile);
057    }
058    return exceptionBuilder;
059  }
060
061  public void resetExceptionBuilder() {
062    exceptionBuilder = null;
063  }
064  /**
065   * Initializes a parser from an input stream. This method is made public for the requirements of IDEs support.
066   *
067   * @param sourceCodeInputStream the source code input stream.
068   * @return the parser.
069   */
070  public final GoloParser initParser(String goloSourceFilename, InputStream sourceCodeInputStream) throws GoloCompilationException {
071    try {
072      return initParser(new InputStreamReader(sourceCodeInputStream, Charset.forName("UTF-8")));
073    } catch (UnsupportedCharsetException e) {
074      getOrCreateExceptionBuilder(goloSourceFilename).report(e).doThrow();
075      return null;
076    }
077  }
078
079  /**
080   * Initializes a parser from a reader. This method is made public for the requirements of IDEs support.
081   *
082   * @param sourceReader the source code reader.
083   * @return the parser.
084   */
085  public final GoloParser initParser(Reader sourceReader) {
086    if (parser == null) {
087      parser = createGoloParser(sourceReader);
088    } else {
089      parser.ReInit(sourceReader);
090    }
091    return parser;
092  }
093
094  /**
095   * Compiles a Golo source file from an input stream, and returns a collection of results.
096   *
097   * @param goloSourceFilename    the source file name.
098   * @param sourceCodeInputStream the source code input stream.
099   * @return a list of compilation results.
100   * @throws GoloCompilationException if a problem occurs during any phase of the compilation work.
101   */
102  public final List<CodeGenerationResult> compile(String goloSourceFilename, InputStream sourceCodeInputStream) throws GoloCompilationException {
103    resetExceptionBuilder();
104    ASTCompilationUnit compilationUnit = parse(goloSourceFilename, initParser(goloSourceFilename, sourceCodeInputStream));
105    GoloModule goloModule = check(compilationUnit);
106    JavaBytecodeGenerationGoloIrVisitor bytecodeGenerator = new JavaBytecodeGenerationGoloIrVisitor();
107    return bytecodeGenerator.generateBytecode(goloModule, goloSourceFilename);
108  }
109
110  private void throwIfErrorEncountered() {
111    if (!getProblems().isEmpty()) {
112      exceptionBuilder.doThrow();
113    }
114  }
115
116  /**
117   * Returns the list of problems encountered during the last compilation
118   *
119   * @return a list of compilation problems.
120   */
121  public List<GoloCompilationException.Problem> getProblems() {
122    if (exceptionBuilder == null) {
123      return Collections.emptyList();
124    }
125    return exceptionBuilder.getProblems();
126  }
127
128  /**
129   * Compiles a Golo source file and writes the resulting JVM bytecode {@code .class} files to a target
130   * folder. The class files are written in a directory structure that respects package names.
131   *
132   * @param goloSourceFilename    the source file name.
133   * @param sourceCodeInputStream the source code input stream.
134   * @param targetFolder          the output target folder.
135   * @throws GoloCompilationException if a problem occurs during any phase of the compilation work.
136   * @throws IOException              if writing the {@code .class} files fails for some reason.
137   */
138  public final void compileTo(String goloSourceFilename, InputStream sourceCodeInputStream, File targetFolder) throws GoloCompilationException, IOException {
139    if (targetFolder.isFile()) {
140      throw new IllegalArgumentException(message("file_exists", targetFolder));
141    }
142    List<CodeGenerationResult> results = compile(goloSourceFilename, sourceCodeInputStream);
143    for (CodeGenerationResult result : results) {
144      File outputFolder = new File(targetFolder, result.getPackageAndClass().packageName().replaceAll("\\.", "/"));
145      if (!outputFolder.exists() && !outputFolder.mkdirs()) {
146        throw new IOException(message("directory_not_created", outputFolder));
147      }
148      File outputFile = new File(outputFolder, result.getPackageAndClass().className() + ".class");
149      try (FileOutputStream out = new FileOutputStream(outputFile)) {
150        out.write(result.getBytecode());
151      }
152    }
153  }
154
155  /**
156   * Compiles a Golo source fila and writes the resulting JVM bytecode {@code .class} files to a Jar file stream.
157   * The class files are written in a directory structure that respects package names.
158   *
159   * @param goloSourceFilename the source file name.
160   * @param sourceCodeInputStream the source code input stream.
161   * @param jarOutputStream the output Jar stream
162   * @throws IOException if writing the {@code .class} files fails for some reason.
163   */
164  public final void compileToJar(String goloSourceFilename, InputStream sourceCodeInputStream, JarOutputStream jarOutputStream) throws IOException {
165    List<CodeGenerationResult> results = compile(goloSourceFilename, sourceCodeInputStream);
166    for (CodeGenerationResult result : results) {
167      String entryName = result.getPackageAndClass().packageName().replaceAll("\\.", "/");
168      if (!entryName.isEmpty()) {
169        entryName = entryName + "/";
170      }
171      entryName = entryName + result.getPackageAndClass().className() + ".class";
172      jarOutputStream.putNextEntry(new ZipEntry(entryName));
173      jarOutputStream.write(result.getBytecode());
174      jarOutputStream.closeEntry();
175    }
176  }
177
178  /**
179   * Produces a parse tree for a Golo source file. This is mostly useful to IDEs.
180   *
181   * @param goloSourceFilename the source file name.
182   * @param parser             the parser to use.
183   * @return the resulting parse tree.
184   * @throws GoloCompilationException if the parser encounters an error.
185   */
186  public final ASTCompilationUnit parse(String goloSourceFilename, GoloParser parser) throws GoloCompilationException {
187    ASTCompilationUnit compilationUnit = null;
188    parser.exceptionBuilder = getOrCreateExceptionBuilder(goloSourceFilename);
189    try {
190      compilationUnit = parser.CompilationUnit();
191      compilationUnit.setFilename(goloSourceFilename);
192    } catch (ParseException pe) {
193      exceptionBuilder.report(pe, compilationUnit);
194    }
195    throwIfErrorEncountered();
196    return compilationUnit;
197  }
198
199  public final ASTCompilationUnit parse(String goloSourceFilename) throws GoloCompilationException, IOException {
200    try (FileInputStream in = new FileInputStream(goloSourceFilename)) {
201      return parse(goloSourceFilename, initParser(goloSourceFilename, in));
202    }
203  }
204
205  /**
206   * Checks that the source code is minimally sound by converting a parse tree to an intermediate representation, and
207   * running a few classic visitors over it. This is mostly useful to IDEs.
208   *
209   * @param compilationUnit the source parse tree.
210   * @return the intermediate representation of the source.
211   * @throws GoloCompilationException if an error exists in the source represented by the input parse tree.
212   */
213  public final GoloModule check(ASTCompilationUnit compilationUnit) {
214    GoloModule goloModule = transform(compilationUnit);
215    return refine(goloModule);
216  }
217
218  public final GoloModule transform(ASTCompilationUnit compilationUnit) {
219    return new ParseTreeToGoloIrVisitor().transform(compilationUnit, exceptionBuilder);
220  }
221
222  public final GoloModule refine(GoloModule goloModule) {
223    if (goloModule != null) {
224      goloModule.accept(new SugarExpansionVisitor());
225      goloModule.accept(new ClosureCaptureGoloIrVisitor());
226      goloModule.accept(new LocalReferenceAssignmentAndVerificationVisitor(exceptionBuilder));
227    }
228    throwIfErrorEncountered();
229    return goloModule;
230  }
231
232  /**
233   * Makes a Golo parser from a reader.
234   *
235   * @param sourceReader the reader.
236   * @return the parser for <code>sourceReader</code>.
237   */
238  protected GoloParser createGoloParser(Reader sourceReader) {
239    return new GoloOffsetParser(sourceReader);
240  }
241}
