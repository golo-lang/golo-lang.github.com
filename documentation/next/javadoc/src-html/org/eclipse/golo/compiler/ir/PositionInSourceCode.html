011package org.eclipse.golo.compiler.ir;
012
013public final class PositionInSourceCode {
014
015  private final int line;
016  private final int column;
017
018  public PositionInSourceCode(int line, int column) {
019    this.line = line;
020    this.column = column;
021  }
022
023  public int getLine() {
024    return line;
025  }
026
027  public int getColumn() {
028    return column;
029  }
030
031  public boolean isNull() {
032    return line == 0 && column == 0;
033  }
034
035  @Override
036  public String toString() {
037    return String.format("{line=%d, column=%d}", line, column);
038  }
039
040  @Override
041  public boolean equals(Object o) {
042    if (this == o) { return true; }
043    if (o == null || getClass() != o.getClass()) { return false; }
044
045    PositionInSourceCode that = (PositionInSourceCode) o;
046    return column == that.column && line == that.line;
047  }
048
049  @Override
050  public int hashCode() {
051    int result = line;
052    result = 31 * result + column;
053    return result;
054  }
055}
