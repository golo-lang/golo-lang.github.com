011package org.eclipse.golo.compiler.ir;
012
013import org.eclipse.golo.compiler.parser.GoloASTNode;
014
015import java.util.List;
016import java.util.LinkedList;
017import static java.util.Collections.unmodifiableList;
018import static java.util.Objects.requireNonNull;
019
020public final class DestructuringAssignment extends GoloAssignment {
021
022  private final List<LocalReference> references = new LinkedList<>();
023  private boolean isVarargs = false;
024
025  DestructuringAssignment() {
026    super();
027  }
028
029  @Override
030  public DestructuringAssignment ofAST(GoloASTNode n) {
031    super.ofAST(n);
032    return this;
033  }
034
035  public boolean isVarargs() {
036    return this.isVarargs;
037  }
038
039  public DestructuringAssignment varargs(boolean varargs) {
040    this.isVarargs = varargs;
041    return this;
042  }
043
044  public DestructuringAssignment varargs() {
045    return varargs(true);
046  }
047
048  @Override
049  public DestructuringAssignment declaring() {
050    return this.declaring(true);
051  }
052
053  @Override
054  public DestructuringAssignment declaring(boolean d) {
055    super.declaring(d);
056    return this;
057  }
058
059  /**
060   * @inheritDoc
061   */
062  @Override
063  public DestructuringAssignment as(Object expr) {
064    super.as(expr);
065    return this;
066  }
067
068  /**
069   * {@inheritDoc}
070   */
071  @Override
072  public LocalReference[] getReferences() {
073    return this.references.toArray(new LocalReference[references.size()]);
074  }
075
076  /**
077   * {@inheritDoc}
078   */
079  @Override
080  public int getReferencesCount() {
081    return this.references.size();
082  }
083
084  /**
085   * @inheritDoc
086   */
087  @Override
088  public DestructuringAssignment to(Object... refs) {
089    for (Object o : refs) {
090      if (o instanceof LocalReference) {
091        references.add((LocalReference) o);
092      } else {
093        throw new IllegalArgumentException("LocalReference expected, got a " + o.getClass().getName());
094      }
095    }
096    return this;
097  }
098
099
100  @Override
101  public String toString() {
102    List<String> names = new LinkedList<>();
103    for (LocalReference r : getReferences()) {
104      names.add(r.toString());
105    }
106    return String.join(", ", names) + " = " + getExpressionStatement().toString();
107  }
108
109  @Override
110  public void accept(GoloIrVisitor visitor) {
111    visitor.visitDestructuringAssignment(this);
112  }
113
114  @Override
115  public void walk(GoloIrVisitor visitor) {
116    for (LocalReference ref : references) {
117      ref.accept(visitor);
118    }
119    super.walk(visitor);
120  }
121}
