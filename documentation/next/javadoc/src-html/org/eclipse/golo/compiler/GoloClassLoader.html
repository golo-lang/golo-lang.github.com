011package org.eclipse.golo.compiler;
012
013import java.io.InputStream;
014import java.util.List;
015
016/**
017 * Provides a facility to dynamically load Golo source code and access the generated code from a dedicated class loader.
018 * <p>
019 * Golo source files can be compiled and the resulting JVM bytecode be injected into the class loader. It is important
020 * to note that this class loader definition is not thread safe.
021 * <p>
022 * This class loader does not support reloading. Attempts to load source files that may produce the same bytecode
023 * definitions will resulting in exceptions.
024 */
025public class GoloClassLoader extends ClassLoader {
026
027  private final GoloCompiler compiler = new GoloCompiler();
028
029  /**
030   * Creates a class loader from a parent.
031   *
032   * @param parent the parent classloader.
033   */
034  public GoloClassLoader(ClassLoader parent) {
035    super(parent);
036  }
037
038  /**
039   * Creates a class loader from the default parent.
040   */
041  public GoloClassLoader() {
042    super();
043  }
044
045  /**
046   * Compiles and loads the resulting JVM bytecode for a Golo source file.
047   *
048   * @param goloSourceFilename    the source file name.
049   * @param sourceCodeInputStream the source input stream.
050   * @return the class matching the Golo module defined in the source.
051   * @throws GoloCompilationException if either of the compilation phase failed.
052   */
053  public synchronized Class<?> load(String goloSourceFilename, InputStream sourceCodeInputStream) throws GoloCompilationException {
054    List<CodeGenerationResult> results = compiler.compile(goloSourceFilename, sourceCodeInputStream);
055    Class<?> lastClassIsModule = null;
056    for (CodeGenerationResult result : results) {
057      byte[] bytecode = result.getBytecode();
058      lastClassIsModule = defineClass(null, bytecode, 0, bytecode.length);
059    }
060    return lastClassIsModule;
061  }
062}
