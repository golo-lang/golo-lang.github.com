011package org.eclipse.golo.compiler.ir;
012
013import org.eclipse.golo.compiler.PackageAndClass;
014
015import java.util.Set;
016import org.eclipse.golo.compiler.parser.GoloASTNode;
017
018import static org.eclipse.golo.compiler.ir.Builders.*;
019
020public final class Struct extends TypeWithMembers {
021
022  public static final String IMMUTABLE_FACTORY_METHOD = "$_immutable";
023
024  private PackageAndClass moduleName;
025
026  @Override
027  public Struct ofAST(GoloASTNode node) {
028    super.ofAST(node);
029    return this;
030  }
031
032  Struct(String name) {
033    super(name);
034  }
035
036  @Override
037  public PackageAndClass getPackageAndClass() {
038    return moduleName.createSubPackage("types").createSubPackage(getName());
039  }
040
041  // TODO: refactor to use the parent node
042  public void setModuleName(PackageAndClass module) {
043    this.moduleName = module;
044  }
045
046  private GoloFunction createDefaultConstructor() {
047    return functionDeclaration(getName()).synthetic().returns(call(getFactoryDelegateName()));
048  }
049
050  private String getImmutableName() {
051    return "Immutable" + getName();
052  }
053
054  private GoloFunction createFullArgsImmutableConstructor() {
055    return functionDeclaration(getImmutableName()).synthetic()
056      .withParameters(getMemberNames())
057      .returns(call(getFactoryDelegateName() + "." + IMMUTABLE_FACTORY_METHOD).withArgs(getFullArgs()));
058  }
059
060
061  public Set<GoloFunction> createFactories() {
062    Set<GoloFunction> factories = super.createFactories();
063    factories.add(createDefaultConstructor());
064    factories.add(createFullArgsImmutableConstructor());
065    return factories;
066  }
067
068  @Override
069  public void accept(GoloIrVisitor visitor) {
070    visitor.visitStruct(this);
071  }
072}
