011package org.eclipse.golo.compiler;
012
013import static java.util.Objects.requireNonNull;
014
015/**
016 * Represents a package and class.
017 */
018public final class PackageAndClass {
019
020  private final String packageName;
021  private final String className;
022
023  /**
024   * Makes a new package and class definition.
025   *
026   * @param packageName the package name.
027   * @param className   the class name.
028   */
029  public PackageAndClass(String packageName, String className) {
030    this.packageName = requireNonNull(packageName);
031    this.className = requireNonNull(className);
032    if (className.isEmpty()) {
033      throw new IllegalArgumentException("The class name can't be empty");
034    }
035  }
036
037  /**
038   * Extracts a package and class definition from a string.
039   *
040   * @param qualifiedName a qualified name.
041   * @return a package and class definition.
042   */
043  public static PackageAndClass fromString(String qualifiedName) {
044    return new PackageAndClass(
045        extractTargetJavaPackage(qualifiedName),
046        extractTargetJavaClass(qualifiedName));
047  }
048
049  private static int packageClassSeparatorIndex(String moduleName) {
050    if (moduleName != null) {
051      return moduleName.lastIndexOf('.');
052    }
053    return -1;
054  }
055
056  private static String extractTargetJavaPackage(String moduleName) {
057    int packageClassSeparatorIndex = packageClassSeparatorIndex(moduleName);
058    if (packageClassSeparatorIndex > 0) {
059      return moduleName.substring(0, packageClassSeparatorIndex);
060    }
061    return "";
062  }
063
064  private static String extractTargetJavaClass(String moduleName) {
065    int packageClassSeparatorIndex = packageClassSeparatorIndex(moduleName);
066    if (packageClassSeparatorIndex > 0) {
067      return moduleName.substring(packageClassSeparatorIndex + 1);
068    }
069    return moduleName;
070  }
071
072
073  /**
074   * Create an inner class.
075   * <p>
076   * For instance:
077   * <pre class="listing"><code class="lang-java" data-lang="java">
078   * PackageAndClass cls = new PackageAndClass("foo.bar", "Spam");
079   * PackageAndClass inner = cls.createInnerClass("Egg"); // foo.bar.Spam$Egg
080   * </code></pre>
081   *
082   * @return a new {@code PackageAndClass} identifying an inner class of this class.
083   */
084  public PackageAndClass createInnerClass(String name) {
085    return new PackageAndClass(
086        this.packageName,
087        this.className + "$" + name.replace('.', '$'));
088  }
089
090  /**
091   * Create a sibling class.
092   * <p>
093   * For instance:
094   * <pre class="listing"><code class="lang-java" data-lang="java">
095   *  PackageAndClass list = new PackageAndClass("java.util", "List");
096   *  PackageAndClass set = list.createSiblingClass("Set"); // java.util.Set
097   * </code></pre>
098   *
099   * @return a new {@code PackageAndClass} identifying an alternate class in the same package.
100   */
101  public PackageAndClass createSiblingClass(String name) {
102    return new PackageAndClass(this.packageName, name);
103  }
104
105  /**
106   * Create a sub-package.
107   * <p>
108   * For instance:
109   * <pre class="listing"><code class="lang-java" data-lang="java">
110   *  PackageAndClass pc = new PackageAndClass("foo.bar", "Module");
111   *  PackageAndClass sub = pc.createSubPackage("types"); // foo.bar.Modules.types
112   * </code></pre>
113   *
114   * @return a new {@code PackageAndClass} identifying a sub-package of this package.
115   */
116  public PackageAndClass createSubPackage(String name) {
117    return new PackageAndClass(
118        this.packageName.isEmpty() ? this.className : this.packageName + "." + this.className, name);
119  }
120
121  /**
122   * Create a class in another package.
123   * <p>
124   * For instance:
125   * <pre class="listing"><code class="lang-java" data-lang="java">
126   * PackageAndClass pc = PackageAndClass.fromString("foo.bar.Baz");
127   * PackageAndClass other = pc.inPackage("plic.ploc"); // plic.ploc.Baz
128   * </code></pre>
129   *
130   * @return a new {@code PackageAndClass} representing the same class in another package.
131   * @param qualifiedName the qualified name of the new package.
132   */
133  public PackageAndClass inPackage(String qualifiedName) {
134    return new PackageAndClass(qualifiedName, className);
135  }
136
137  /**
138   * Create a class in the same package as another one.
139   * <p>
140   * For instance:
141   * <pre class="listing"><code class="lang-java" data-lang="java">
142   * PackageAndClass pc = PackageAndClass.fromString("foo.bar.Baz");
143   * PackageAndClass other = PackageAndClass.fromString("plic.ploc.Foo");
144   * PackageAndClass newOne = pc.inPackage(other); // plic.ploc.Baz
145   * </code></pre>
146   *
147   * @return a new {@code PackageAndClass} representing the same class in another package.
148   * @param parent the {@code PackageAndClass} representing the new package.
149   */
150  public PackageAndClass inPackage(PackageAndClass parent) {
151    return new PackageAndClass(parent.packageName(), className);
152  }
153
154  /**
155   * @return the package name.
156   */
157  public String packageName() {
158    return packageName;
159  }
160
161  /**
162   * Check if this {@code PackageAndClass} has a package component.
163   */
164  public boolean hasPackage() {
165    return !packageName.isEmpty();
166  }
167
168  /**
169   * @return the package as a {@code PackageAndClass}
170   */
171  public PackageAndClass parentPackage() {
172    return fromString(packageName);
173  }
174
175  /**
176   * @return the class name.
177   */
178  public String className() {
179    return className;
180  }
181
182  @Override
183  public String toString() {
184    if (packageName.isEmpty()) {
185      return className;
186    } else {
187      return packageName + "." + className;
188    }
189  }
190
191  /**
192   * @return a JVM type representation for this object, e.g.: {@code foo.Bar} gives {@code foo/Bar}.
193   */
194  public String toJVMType() {
195    return toString().replaceAll("\\.", "/");
196  }
197
198  /**
199   * @return a JVM reference type representation for this object, e.g.: {@code foo.Bar} gives
200   * {@code Lfoo/Bar;}
201   */
202  public String toJVMRef() {
203    return "L" + toJVMType() + ";";
204  }
205
206  /**
207   * @return a mangled named for this class.
208   */
209  public String mangledName() {
210    return toString().replace('.', '$');
211  }
212
213  @Override
214  public boolean equals(Object o) {
215    if (this == o) return true;
216    if (o == null || getClass() != o.getClass()) return false;
217
218    PackageAndClass that = (PackageAndClass) o;
219
220    if (className != null ? !className.equals(that.className) : that.className != null) return false;
221    if (packageName != null ? !packageName.equals(that.packageName) : that.packageName != null) return false;
222
223    return true;
224  }
225
226  @Override
227  public int hashCode() {
228    int result = packageName != null ? packageName.hashCode() : 0;
229    result = 31 * result + (className != null ? className.hashCode() : 0);
230    return result;
231  }
232}
