011package org.eclipse.golo.cli.command;
012
013import com.beust.jcommander.IParameterValidator;
014import com.beust.jcommander.Parameter;
015import com.beust.jcommander.ParameterException;
016import com.beust.jcommander.Parameters;
017import org.eclipse.golo.cli.command.spi.CliCommand;
018import org.eclipse.golo.compiler.GoloCompiler;
019import org.eclipse.golo.compiler.GoloCompilationException;
020import org.eclipse.golo.doc.AbstractProcessor;
021import org.eclipse.golo.doc.CtagsProcessor;
022import org.eclipse.golo.doc.HtmlProcessor;
023import org.eclipse.golo.doc.MarkdownProcessor;
024import org.eclipse.golo.doc.ModuleDocumentation;
025
026import java.io.File;
027import java.io.IOException;
028import java.nio.file.Paths;
029import java.util.*;
030
031import static gololang.Messages.*;
032
033@Parameters(commandNames = {"doc"}, commandDescriptionKey = "doc", resourceBundle = "commands")
034public class DocCommand implements CliCommand {
035
036  @Parameter(names = "--format", descriptionKey = "doc.format", validateWith = DocFormatValidator.class)
037  String format = "html";
038
039  @Parameter(names = "--output", descriptionKey = "doc.output")
040  String output = ".";
041
042  @Parameter(descriptionKey = "source_files")
043  List<String> sources = new LinkedList<>();
044
045  private static final Map<String, AbstractProcessor> FORMATS = new HashMap<>();
046  static {
047    // TODO: use a service provider ?
048    FORMATS.put("markdown", new MarkdownProcessor());
049    FORMATS.put("html", new HtmlProcessor());
050    FORMATS.put("ctags", new CtagsProcessor());
051  }
052
053  private final GoloCompiler compiler = new GoloCompiler();
054
055  @Override
056  public void execute() throws Throwable {
057
058    AbstractProcessor processor = FORMATS.get(this.format);
059    HashMap<String, ModuleDocumentation> modules = new HashMap<>();
060    for (String source : this.sources) {
061      loadGoloFile(source, modules);
062    }
063    try {
064      processor.process(modules, Paths.get(this.output));
065    } catch (Throwable throwable) {
066      handleThrowable(throwable);
067    }
068  }
069
070  private void loadGoloFile(String goloFile, HashMap<String, ModuleDocumentation> modules) {
071    File file = new File(goloFile);
072    if (file.isDirectory()) {
073      File[] directoryFiles = file.listFiles();
074      if (directoryFiles != null) {
075        for (File directoryFile : directoryFiles) {
076          loadGoloFile(directoryFile.getAbsolutePath(), modules);
077        }
078      }
079    } else if (file.getName().endsWith(".golo")) {
080      try {
081        modules.put(goloFile, ModuleDocumentation.load(goloFile, compiler));
082      } catch (IOException e) {
083        error(message("file_not_found", goloFile));
084        return;
085      } catch (GoloCompilationException e) {
086        handleCompilationException(e);
087      }
088    }
089  }
090
091  public static class DocFormatValidator implements IParameterValidator {
092
093    @Override
094    public void validate(String name, String value) throws ParameterException {
095      if (!FORMATS.keySet().contains(value)) {
096        throw new ParameterException(message("format_error", FORMATS.keySet()));
097      }
098    }
099  }
100}
